30LIB8STATIC_ALWAYS_INLINE uint8_t scale8( uint8_t i, fract8 scale)
31{
32#if SCALE8_C == 1
33#if (FASTLED_SCALE8_FIXED == 1)
34 return (((uint16_t)i) * (1+(uint16_t)(scale))) >> 8;
35#else
36 return ((uint16_t)i * (uint16_t)(scale) ) >> 8;
37#endif
38#elif SCALE8_AVRASM == 1
39#if defined(LIB8_ATTINY)
40#if (FASTLED_SCALE8_FIXED == 1)
41 uint8_t work=i;
42#else
43 uint8_t work=0;
44#endif
45 uint8_t cnt=0x80;
46 asm volatile(
47#if (FASTLED_SCALE8_FIXED == 1)
48 " inc %[scale] \n\t"
49 " breq DONE_%= \n\t"
50 " clr %[work] \n\t"
51#endif
52 "LOOP_%=: \n\t"
53 /*" sbrc %[scale], 0 \n\t"
54 " add %[work], %[i] \n\t"
55 " ror %[work] \n\t"
56 " lsr %[scale] \n\t"
57 " clc \n\t"*/
58 " sbrc %[scale], 0 \n\t"
59 " add %[work], %[i] \n\t"
60 " ror %[work] \n\t"
61 " lsr %[scale] \n\t"
62 " lsr %[cnt] \n\t"
63 "brcc LOOP_%= \n\t"
64 "DONE_%=: \n\t"
65 : [work] "+r" (work), [cnt] "+r" (cnt)
66 : [scale] "r" (scale), [i] "r" (i)
67 :
68 );
69 return work;
70#else
71 asm volatile(
72#if (FASTLED_SCALE8_FIXED==1)
73 // Multiply 8-bit i * 8-bit scale, giving 16-bit r1,r0
74 "mul %0, %1 \n\t"
75 // Add i to r0, possibly setting the carry flag
76 "add r0, %0 \n\t"
77 // load the immediate 0 into i (note, this does _not_ touch any flags)
78 "ldi %0, 0x00 \n\t"
79 // walk and chew gum at the same time
80 "adc %0, r1 \n\t"
81#else
82 /* Multiply 8-bit i * 8-bit scale, giving 16-bit r1,r0 */
83 "mul %0, %1 \n\t"
84 /* Move the high 8-bits of the product (r1) back to i */
85 "mov %0, r1 \n\t"
86 /* Restore r1 to "0"; it's expected to always be that */
87#endif
88 "clr __zero_reg__ \n\t"
89
90 : "+d" (i) /* writes to i; r16-r31, restricted by ldi */
91 : "r" (scale) /* uses scale */
92 : "r0", "r1" /* clobbers r0, r1 */
93 );
94 /* Return the result */
95 return i;
96#endif
97#else
98#error "No implementation for scale8 available."
99#endif

476 }
477#if SCALE16BY8_C == 1

485#elif SCALE16BY8_AVRASM == 1

546LIB8STATIC uint16_t scale16( uint16_t i, fract16 scale )
547{
548 #if SCALE16_C == 1
549 uint16_t result;
550#if FASTLED_SCALE8_FIXED == 1
551 result = ((uint32_t)(i) * (1+(uint32_t)(scale))) / 65536;
552#else
553 result = ((uint32_t)(i) * (uint32_t)(scale)) / 65536;
554#endif
555 return result;
556#elif SCALE16_AVRASM == 1
557#if FASTLED_SCALE8_FIXED == 1
558 // implemented sort of like
559 // result = ((i * scale) + i ) / 65536
560 //
561 // why not like this, you may ask?
562 // result = (i * (scale+1)) / 65536
563 // the answer is that if scale is 65535, then scale+1
564 // will be zero, which is not what we want.
565 uint32_t result;
566 asm volatile(
567 // result.A-B = i.A x scale.A
568 " mul %A[i], %A[scale] \n\t"
569 // save results...
570 // basic idea:
571 //" mov %A[result], r0 \n\t"
572 //" mov %B[result], r1 \n\t"
573 // which can be written as...
574 " movw %A[result], r0 \n\t"
575 // Because we're going to add i.A-B to
576 // result.A-D, we DO need to keep both
577 // the r0 and r1 portions of the product
578 // UNlike in the 'unfixed scale8' version.
579 // So the movw here is needed.
580 : [result] "=r" (result)
581 : [i] "r" (i),
582 [scale] "r" (scale)
583 : "r0", "r1"
584 );
585
586 asm volatile(
587 // result.C-D = i.B x scale.B
588 " mul %B[i], %B[scale] \n\t"
589 //" mov %C[result], r0 \n\t"
590 //" mov %D[result], r1 \n\t"
591 " movw %C[result], r0 \n\t"
592 : [result] "+r" (result)
593 : [i] "r" (i),
594 [scale] "r" (scale)
595 : "r0", "r1"
596);
597
598const uint8_t zero = 0;
599asm volatile(
600 // result.B-D += i.B x scale.A
601 " mul %B[i], %A[scale] \n\t"
602
603 " add %B[result], r0 \n\t"
604 " adc %C[result], r1 \n\t"
605 " adc %D[result], %[zero] \n\t"
606
607 // result.B-D += i.A x scale.B
608 " mul %A[i], %B[scale] \n\t"
609
610 " add %B[result], r0 \n\t"
611 " adc %C[result], r1 \n\t"
612 " adc %D[result], %[zero] \n\t"
613
614 // cleanup r1
615 " clr r1 \n\t"
616
617 : [result] "+r" (result)
618 : [i] "r" (i),
619 [scale] "r" (scale),
620 [zero] "r" (zero)
621 : "r0", "r1"
622 );
623
624 asm volatile(
625 // result.A-D += i.A-B
626 " add %A[result], %A[i] \n\t"
627 " adc %B[result], %B[i] \n\t"
628 " adc %C[result], %[zero] \n\t"
629 " adc %D[result], %[zero] \n\t"
630 : [result] "+r" (result)
631 : [i] "r" (i),
632 [zero] "r" (zero)
633 );
634
635 result = result >> 16;
636 return result;
637#else
638 uint32_t result;
639 asm volatile(
640 // result.A-B = i.A x scale.A
641 " mul %A[i], %A[scale] \n\t"
642 // save results...
643 // basic idea:
644 //" mov %A[result], r0 \n\t"
645 //" mov %B[result], r1 \n\t"
646 // which can be written as...
647 " movw %A[result], r0 \n\t"
648 // We actually don't need to do anything with r0,
649 // as result.A is never used again here, so we
650 // could just move the high byte, but movw is
651 // one clock cycle, just like mov, so might as
652 // well, in case we want to use this code for
653 // a generic 16x16 multiply somewhere.
654
655 : [result] "=r" (result)
656 : [i] "r" (i),
657 [scale] "r" (scale)
658 : "r0", "r1"
659 );
660
661 asm volatile(
662 // result.C-D = i.B x scale.B
663 " mul %B[i], %B[scale] \n\t"
664 //" mov %C[result], r0 \n\t"
665 //" mov %D[result], r1 \n\t"
666 " movw %C[result], r0 \n\t"
667 : [result] "+r" (result)
668 : [i] "r" (i),
669 [scale] "r" (scale)
670 : "r0", "r1"
671 );
672
673 const uint8_t zero = 0;
674 asm volatile(
675 // result.B-D += i.B x scale.A
676 " mul %B[i], %A[scale] \n\t"
677
678 " add %B[result], r0 \n\t"
679 " adc %C[result], r1 \n\t"
680 " adc %D[result], %[zero] \n\t"
681

683 " mul %A[i], %B[scale] \n\t"
