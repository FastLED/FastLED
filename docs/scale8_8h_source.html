29LIB8STATIC_ALWAYS_INLINE uint8_t scale8(uint8_t i, fract8 scale) {
30#if SCALE8_C == 1
31#if (FASTLED_SCALE8_FIXED == 1)
32 return (((uint16_t)i) * (1 + (uint16_t)(scale))) >> 8;
33#else
34 return ((uint16_t)i * (uint16_t)(scale)) >> 8;
35#endif
36#elif SCALE8_AVRASM == 1
37#if defined(LIB8_ATTINY)
38#if (FASTLED_SCALE8_FIXED == 1)
39 uint8_t work = i;
40#else
41 uint8_t work = 0;
42#endif
43 uint8_t cnt = 0x80;
44 asm volatile(
45#if (FASTLED_SCALE8_FIXED == 1)
46 " inc %[scale] \n\t"
47 " breq DONE_%= \n\t"
48 " clr %[work] \n\t"
49#endif
50 "LOOP_%=: \n\t"
51 /*" sbrc %[scale], 0 \n\t"
52 " add %[work], %[i] \n\t"
53 " ror %[work] \n\t"
54 " lsr %[scale] \n\t"
55 " clc \n\t"*/
56 " sbrc %[scale], 0 \n\t"
57 " add %[work], %[i] \n\t"
58 " ror %[work] \n\t"
59 " lsr %[scale] \n\t"
60 " lsr %[cnt] \n\t"
61 "brcc LOOP_%= \n\t"
62 "DONE_%=: \n\t"
63 : [work] "+r"(work), [cnt] "+r"(cnt)
64 : [scale] "r"(scale), [i] "r"(i)
65 :);
66 return work;
67#else
68 asm volatile(
69#if (FASTLED_SCALE8_FIXED == 1)
70 // Multiply 8-bit i * 8-bit scale, giving 16-bit r1,r0
71 "mul %0, %1 \n\t"
72 // Add i to r0, possibly setting the carry flag
73 "add r0, %0 \n\t"
74 // load the immediate 0 into i (note, this does _not_ touch any flags)
75 "ldi %0, 0x00 \n\t"
76 // walk and chew gum at the same time
77 "adc %0, r1 \n\t"
78#else
79 /* Multiply 8-bit i * 8-bit scale, giving 16-bit r1,r0 */
80 "mul %0, %1 \n\t"
81 /* Move the high 8-bits of the product (r1) back to i */
82 "mov %0, r1 \n\t"
83 /* Restore r1 to "0"; it's expected to always be that */
84#endif
85 "clr __zero_reg__ \n\t"
86
87 : "+d"(i) /* writes to i; r16-r31, restricted by ldi */
88 : "r"(scale) /* uses scale */
89 : "r0", "r1" /* clobbers r0, r1 */
90 );
91 /* Return the result */
92 return i;
93#endif
94#else
95#error "No implementation for scale8 available."

526LIB8STATIC uint16_t scale16(uint16_t i, fract16 scale) {
527#if SCALE16_C == 1
528 uint16_t result;
529#if FASTLED_SCALE8_FIXED == 1
530 result = ((uint32_t)(i) * (1 + (uint32_t)(scale))) / 65536;
531#else
532 result = ((uint32_t)(i) * (uint32_t)(scale)) / 65536;
533#endif
534 return result;
535#elif SCALE16_AVRASM == 1
536#if FASTLED_SCALE8_FIXED == 1
537 // implemented sort of like
538 // result = ((i * scale) + i ) / 65536
539 //
540 // why not like this, you may ask?
541 // result = (i * (scale+1)) / 65536
542 // the answer is that if scale is 65535, then scale+1
543 // will be zero, which is not what we want.
544 uint32_t result;
545 asm volatile(
546 // result.A-B = i.A x scale.A
547 " mul %A[i], %A[scale] \n\t"
548 // save results...
549 // basic idea:
550 //" mov %A[result], r0 \n\t"
551 //" mov %B[result], r1 \n\t"
552 // which can be written as...
553 " movw %A[result], r0 \n\t"
554 // Because we're going to add i.A-B to
555 // result.A-D, we DO need to keep both
556 // the r0 and r1 portions of the product
557 // UNlike in the 'unfixed scale8' version.
558 // So the movw here is needed.
559 : [result] "=r"(result)
560 : [i] "r"(i), [scale] "r"(scale)
561 : "r0", "r1");
562
563 asm volatile(
564 // result.C-D = i.B x scale.B
565 " mul %B[i], %B[scale] \n\t"
566 //" mov %C[result], r0 \n\t"
567 //" mov %D[result], r1 \n\t"
568 " movw %C[result], r0 \n\t"
569 : [result] "+r"(result)
570 : [i] "r"(i), [scale] "r"(scale)
571 : "r0", "r1");
572
573 const uint8_t zero = 0;
574 asm volatile(
575 // result.B-D += i.B x scale.A
576 " mul %B[i], %A[scale] \n\t"
577
578 " add %B[result], r0 \n\t"
579 " adc %C[result], r1 \n\t"
580 " adc %D[result], %[zero] \n\t"
581
582 // result.B-D += i.A x scale.B
583 " mul %A[i], %B[scale] \n\t"
584
585 " add %B[result], r0 \n\t"
586 " adc %C[result], r1 \n\t"

605 result = result >> 16;
606 return result;
607#else
608 uint32_t result;

610 // result.A-B = i.A x scale.A
611 " mul %A[i], %A[scale] \n\t"
612 // save results...
613 // basic idea:
614 //" mov %A[result], r0 \n\t"
615 //" mov %B[result], r1 \n\t"
616 // which can be written as...

621 // one clock cycle, just like mov, so might as
