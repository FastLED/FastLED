150 template <uint8_t BIT> __attribute__((always_inline, hot)) inline static void writeBit(uint8_t b) {
151 //cli();
152 if(b & (1 << BIT)) {
153 FastPin<DATA_PIN>::hi();
154#ifdef ESP32
155 // try to ensure we never have adjacent write opcodes to the same register
156 FastPin<CLOCK_PIN>::lo();
157 FastPin<CLOCK_PIN>::hi(); CLOCK_HI_DELAY;
158 FastPin<CLOCK_PIN>::toggle(); CLOCK_LO_DELAY;
159#else
160 FastPin<CLOCK_PIN>::hi(); CLOCK_HI_DELAY;
161 FastPin<CLOCK_PIN>::lo(); CLOCK_LO_DELAY;
162#endif
163 } else {
164 FastPin<DATA_PIN>::lo();
165 FastPin<CLOCK_PIN>::hi(); CLOCK_HI_DELAY;
166#ifdef ESP32
167 // try to ensure we never have adjacent write opcodes to the same register
168 FastPin<CLOCK_PIN>::toggle(); CLOCK_HI_DELAY;
169#else
170 FastPin<CLOCK_PIN>::lo(); CLOCK_LO_DELAY;
171#endif
172 }

178 template <uint8_t BIT> __attribute__((always_inline)) inline static void writeBit(uint8_t b, clock_ptr_t clockpin, data_ptr_t datapin) {
179 if(b & (1 << BIT)) {
180 FastPin<DATA_PIN>::hi(datapin);
181 FastPin<CLOCK_PIN>::hi(clockpin); CLOCK_HI_DELAY;
182 FastPin<CLOCK_PIN>::lo(clockpin); CLOCK_LO_DELAY;
183 } else {
184 FastPin<DATA_PIN>::lo(datapin);
185 FastPin<CLOCK_PIN>::hi(clockpin); CLOCK_HI_DELAY;
186 FastPin<CLOCK_PIN>::lo(clockpin); CLOCK_LO_DELAY;
187 }

253 static void writeBytesValueRaw(uint8_t value, int len) {
254#ifdef FAST_SPI_INTERRUPTS_WRITE_PINS
255 // TODO: Weird things may happen if software bitbanging SPI output and other pins on the output reigsters are being twiddled. Need
256 // to allow specifying whether or not exclusive i/o access is allowed during this process, and if i/o access is not allowed fall
257 // back to the degenerative code below
258 while(len--) {
259 writeByte(value);
260 }
261#else
262 FASTLED_REGISTER data_ptr_t datapin = FastPin<DATA_PIN>::port();
263
264 if(FastPin<DATA_PIN>::port() != FastPin<CLOCK_PIN>::port()) {
265 // If data and clock are on different ports, then writing a bit will consist of writing the value foor
266 // the bit (hi or low) to the data pin port, and then two writes to the clock port to strobe the clock line
267 FASTLED_REGISTER clock_ptr_t clockpin = FastPin<CLOCK_PIN>::port();
268 FASTLED_REGISTER data_t datahi = FastPin<DATA_PIN>::hival();
269 FASTLED_REGISTER data_t datalo = FastPin<DATA_PIN>::loval();
270 FASTLED_REGISTER clock_t clockhi = FastPin<CLOCK_PIN>::hival();
271 FASTLED_REGISTER clock_t clocklo = FastPin<CLOCK_PIN>::loval();
272 while(len--) {
273 writeByte(value, clockpin, datapin, datahi, datalo, clockhi, clocklo);
274 }
275
276 } else {
277 // If data and clock are on the same port then we can combine setting the data and clock pins
278 FASTLED_REGISTER data_t datahi_clockhi = FastPin<DATA_PIN>::hival() | FastPin<CLOCK_PIN>::mask();
279 FASTLED_REGISTER data_t datalo_clockhi = FastPin<DATA_PIN>::loval() | FastPin<CLOCK_PIN>::mask();
280 FASTLED_REGISTER data_t datahi_clocklo = FastPin<DATA_PIN>::hival() & ~FastPin<CLOCK_PIN>::mask();
281 FASTLED_REGISTER data_t datalo_clocklo = FastPin<DATA_PIN>::loval() & ~FastPin<CLOCK_PIN>::mask();
282
283 while(len--) {
284 writeByte(value, datapin, datahi_clockhi, datalo_clockhi, datahi_clocklo, datalo_clocklo);
285 }
286 }
287#endif
288 }

351 template <uint8_t FLAGS, class D, EOrder RGB_ORDER> __attribute__((noinline)) void writePixels(PixelController<RGB_ORDER> pixels) {
352 select();
353 int len = pixels.mLen;
354
355#ifdef FAST_SPI_INTERRUPTS_WRITE_PINS
356 // If interrupts or other things may be generating output while we're working on things, then we need
357 // to use this block
358 while(pixels.has(1)) {
359 if(FLAGS & FLAG_START_BIT) {
360 writeBit<0>(1);
361 }
362 writeByte(D::adjust(pixels.loadAndScale0()));
363 writeByte(D::adjust(pixels.loadAndScale1()));
364 writeByte(D::adjust(pixels.loadAndScale2()));
365 pixels.advanceData();
366 pixels.stepDithering();
367 }
368#else
369 // If we can guaruntee that no one else will be writing data while we are running (namely, changing the values of the PORT/PDOR pins)
370 // then we can use a bunch of optimizations in here
371 FASTLED_REGISTER data_ptr_t datapin = FastPin<DATA_PIN>::port();
372
373 if(FastPin<DATA_PIN>::port() != FastPin<CLOCK_PIN>::port()) {
374 FASTLED_REGISTER clock_ptr_t clockpin = FastPin<CLOCK_PIN>::port();
375 // If data and clock are on different ports, then writing a bit will consist of writing the value foor
376 // the bit (hi or low) to the data pin port, and then two writes to the clock port to strobe the clock line
377 FASTLED_REGISTER data_t datahi = FastPin<DATA_PIN>::hival();
378 FASTLED_REGISTER data_t datalo = FastPin<DATA_PIN>::loval();
379 FASTLED_REGISTER clock_t clockhi = FastPin<CLOCK_PIN>::hival();
380 FASTLED_REGISTER clock_t clocklo = FastPin<CLOCK_PIN>::loval();
381
382 while(pixels.has(1)) {
383 if(FLAGS & FLAG_START_BIT) {
384 writeBit<0>(1, clockpin, datapin, datahi, datalo, clockhi, clocklo);
385 }
386 writeByte(D::adjust(pixels.loadAndScale0()), clockpin, datapin, datahi, datalo, clockhi, clocklo);
387 writeByte(D::adjust(pixels.loadAndScale1()), clockpin, datapin, datahi, datalo, clockhi, clocklo);
388 writeByte(D::adjust(pixels.loadAndScale2()), clockpin, datapin, datahi, datalo, clockhi, clocklo);
389 pixels.advanceData();
390 pixels.stepDithering();
391 }
392
393 } else {
394 // If data and clock are on the same port then we can combine setting the data and clock pins
395 FASTLED_REGISTER data_t datahi_clockhi = FastPin<DATA_PIN>::hival() | FastPin<CLOCK_PIN>::mask();
396 FASTLED_REGISTER data_t datalo_clockhi = FastPin<DATA_PIN>::loval() | FastPin<CLOCK_PIN>::mask();
397 FASTLED_REGISTER data_t datahi_clocklo = FastPin<DATA_PIN>::hival() & ~FastPin<CLOCK_PIN>::mask();
398 FASTLED_REGISTER data_t datalo_clocklo = FastPin<DATA_PIN>::loval() & ~FastPin<CLOCK_PIN>::mask();
399
400 while(pixels.has(1)) {
401 if(FLAGS & FLAG_START_BIT) {
402 writeBit<0>(1, datapin, datahi_clockhi, datalo_clockhi, datahi_clocklo, datalo_clocklo);
403 }
404 writeByte(D::adjust(pixels.loadAndScale0()), datapin, datahi_clockhi, datalo_clockhi, datahi_clocklo, datalo_clocklo);
405 writeByte(D::adjust(pixels.loadAndScale1()), datapin, datahi_clockhi, datalo_clockhi, datahi_clocklo, datalo_clocklo);
406 writeByte(D::adjust(pixels.loadAndScale2()), datapin, datahi_clockhi, datalo_clockhi, datahi_clocklo, datalo_clocklo);
407 pixels.advanceData();

411#endif
