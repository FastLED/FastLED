65void five_bit_hd_gamma_bitshift(
66 uint8_t r8, uint8_t g8, uint8_t b8,
67 uint8_t r8_scale, uint8_t g8_scale, uint8_t b8_scale,
68 uint8_t* out_r8,
69 uint8_t* out_g8,
70 uint8_t* out_b8,
71 uint8_t* out_power_5bit) {
72
73 // Step 1: Gamma Correction
74 uint16_t r16, g16, b16;
75 five_bit_hd_gamma_function(r8, g8, b8, &r16, &g16, &b16);
76
77 // Step 2: Post gamma correction scale.
78 if (r8_scale != 0xff || g8_scale != 0xff || b8_scale != 0xff) {
79 r16 = scale16by8(r16, r8_scale);
80 g16 = scale16by8(g16, g8_scale);
81 b16 = scale16by8(b16, b8_scale);
82 }
83
84 // Step 3: Initialize 5-bit brightness.
85 // Note: we only get 5 levels of brightness
86 uint8_t v8 = 31;
87
88 uint16_t numerator = 1;
89 uint16_t denominator = 1;
90 const uint32_t r16_const = r16;
91 const uint32_t g16_const = g16;
92 const uint32_t b16_const = b16;
93
94 // Step 4: Bit Shifting Loop, can probably replaced with a
95 // single pass bit-twiddling hack.
96 do {
97 // Note that to avoid slow divisions, we multiply the max_value
98 // by the denominator.
99 uint32_t max_value = 0xfffful * 15;
100 if (r16_const * 31 > max_value) {
101 break;
102 }
103 if (g16_const * 31 > max_value) {
104 break;
105 }
106 if (b16_const * 31 > max_value) {
107 break;
108 }
109 numerator = 31;
110 denominator = 15;
111 v8 = 15;
112
113 max_value = 0xfffful * 15 * 7;
114 if (r16_const * 31 * 15 > max_value) {
115 break;
116 }
117 if (g16_const * 31 * 15 > max_value) {
118 break;
119 }
120 if (b16_const * 31 * 15 > max_value) {
121 break;
122 }
123 numerator = 31 * 15;
124 denominator = 15 * 7;
125 v8 = 7;
126
127 max_value = 0xfffful * 15 * 7 * 3;
128 if (r16_const * 31 * 15 * 7 > max_value) {
129 break;
130 }
131 if (g16_const * 31 * 15 * 7 > max_value) {
132 break;
133 }
134 if (b16_const * 31 * 15 * 7 > max_value) {
135 break;
136 }
137 numerator = 31 * 15 * 7;
138 denominator = 15 * 7 * 3;
139 v8 = 3;
140
141 max_value = 0xfffful * 15 * 7 * 3;
142 if (r16_const * 31 * 15 * 7 * 3 > max_value) {
143 break;
144 }
145 if (g16_const * 31 * 15 * 7 * 3 > max_value) {
146 break;
147 }
148 if (b16_const * 31 * 15 * 7 * 3 > max_value) {
149 break;
150 }
151 numerator = 31 * 15 * 7 * 3;
152 v8 = 1;
153 } while(false);
154
155 r16 = uint16_t(r16_const * numerator / denominator);
156 g16 = uint16_t(g16_const * numerator / denominator);
157 b16 = uint16_t(b16_const * numerator / denominator);
158
159 // Step 5: Conversion Back to 8-bit.
160 uint8_t r8_final = (r8 == 255 && uint8_t(r16 >> 8) >= 254) ? 255 : uint8_t(r16 >> 8);
161 uint8_t g8_final = (g8 == 255 && uint8_t(g16 >> 8) >= 254) ? 255 : uint8_t(g16 >> 8);
162 uint8_t b8_final = (b8 == 255 && uint8_t(b16 >> 8) >= 254) ? 255 : uint8_t(b16 >> 8);
163
164#if FASTLED_FIVE_BIT_HD_GAMMA_LOW_END_LINEAR_RAMP > 0
165 if (v8 == 1) {
166 // Linear tuning for the lowest possible brightness. x=y until
167 // the intersection point at 9.
168 if (r8 < FASTLED_FIVE_BIT_HD_GAMMA_LOW_END_LINEAR_RAMP && r16 > 0) {
169 r8_final = r8;
170 }
171 if (g8 < FASTLED_FIVE_BIT_HD_GAMMA_LOW_END_LINEAR_RAMP && g16 > 0) {
172 g8_final = g8;
173 }
174 if (b8 < FASTLED_FIVE_BIT_HD_GAMMA_LOW_END_LINEAR_RAMP && b16 > 0) {
175 b8_final = b8;
176 }
177 }
178#endif
179
180 // Step 6: Output
181 *out_r8 = r8_final;
182 *out_g8 = g8_final;
183 *out_b8 = b8_final;
184 *out_power_5bit = v8;
185}
