67void app_main(void)
68{
69 led_strip_handle_t led_strip = configure_led();
70 bool led_on_off = false;
71
72 ESP_LOGI(TAG, "Start blinking LED strip");
73 while (1) {
74 if (led_on_off) {
75 /* Set the LED pixel using RGB from 0 (0%) to 255 (100%) for each color */
76 for (int i = 0; i < LED_STRIP_LED_NUMBERS; i++) {
77 ESP_ERROR_CHECK(led_strip_set_pixel(led_strip, i, 5, 5, 5));
78 }
79 /* Refresh the strip to send data */
80 ESP_ERROR_CHECK(led_strip_refresh(led_strip));
81 ESP_LOGI(TAG, "LED ON!");
82 } else {
83 /* Set all LED off to clear all pixels */
84 ESP_ERROR_CHECK(led_strip_clear(led_strip));
85 ESP_LOGI(TAG, "LED OFF!");
86 }
87
88 led_on_off = !led_on_off;
89 vTaskDelay(pdMS_TO_TICKS(500));
90 }
91}
