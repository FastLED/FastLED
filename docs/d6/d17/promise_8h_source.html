237enum class PromiseState_t {
238 PENDING, // Promise is still pending
239 RESOLVED, // Promise completed successfully
240 REJECTED // Promise completed with error
241};
242
244template<typename T>
245class PromiseImpl {
246public:
247 PromiseImpl() FL_NOEXCEPT : mState(static_cast<int>(PromiseState_t::PENDING)), mCallbacksProcessed(false) {}
248
250 void set_then_callback(fl::function<void(const T&)> callback) FL_NOEXCEPT {
251 mThenCallback = fl::move(callback);
252 // If already resolved, process callback immediately
253 if (state() == PromiseState_t::RESOLVED && !mCallbacksProcessed) {
254 process_callbacks();
255 }
256 }
257
259 void set_catch_callback(fl::function<void(const Error&)> callback) FL_NOEXCEPT {
260 mCatchCallback = fl::move(callback);
261 // If already rejected, process callback immediately
262 if (state() == PromiseState_t::REJECTED && !mCallbacksProcessed) {
263 process_callbacks();
264 }
265 }
266
268 void update() FL_NOEXCEPT {
269 // Process callbacks if we're completed and haven't processed them yet
270 if (is_completed() && !mCallbacksProcessed) {
271 process_callbacks();
272 }
273 }
274
276 bool resolve(const T& value) FL_NOEXCEPT {
277 if (state() != PromiseState_t::PENDING) return false;
278
279 // Write value BEFORE setting state — the atomic store provides
280 // a release fence so the reader sees the value after observing
281 // the state transition.
282 mValue = value;
283 set_state(PromiseState_t::RESOLVED);
284
285 // Process callback immediately if we have one
286 if (mThenCallback && !mCallbacksProcessed) {
287 process_callbacks();
288 }
289
290 return true;
291 }
292
293 bool resolve(T&& value) FL_NOEXCEPT {
294 if (state() != PromiseState_t::PENDING) return false;
295
296 mValue = fl::move(value);
297 set_state(PromiseState_t::RESOLVED);
298
299 if (mThenCallback && !mCallbacksProcessed) {
300 process_callbacks();
301 }
302
303 return true;
304 }
305
307 bool reject(const Error& error) FL_NOEXCEPT {
308 if (state() != PromiseState_t::PENDING) return false;
309
310 mError = error;
311 set_state(PromiseState_t::REJECTED);
312
313 if (mCatchCallback && !mCallbacksProcessed) {
314 process_callbacks();
315 }
316
317 return true;
318 }
319
321 bool is_completed() const FL_NOEXCEPT {
322 return state() != PromiseState_t::PENDING;
323 }
324
326 bool is_resolved() const FL_NOEXCEPT {
327 return state() == PromiseState_t::RESOLVED;
328 }
329
331 bool is_rejected() const FL_NOEXCEPT {
332 return state() == PromiseState_t::REJECTED;
333 }
334
336 const T& value() const FL_NOEXCEPT {
337 return mValue;
338 }
339
341 const Error& error() const FL_NOEXCEPT {
342 return mError;
343 }
344
345private:
346 fl::atomic<int> mState; // stores PromiseState_t as int (atomics require integer type)
347 T mValue;
348 Error mError;
349
350 fl::function<void(const T&)> mThenCallback;
351 fl::function<void(const Error&)> mCatchCallback;
352
353 bool mCallbacksProcessed;
354
356 PromiseState_t state() const FL_NOEXCEPT {
357 return static_cast<PromiseState_t>(mState.load());
358 }
359
361 void set_state(PromiseState_t s) FL_NOEXCEPT {
362 mState.store(static_cast<int>(s));
363 }
364
366 void process_callbacks() FL_NOEXCEPT {
367 if (mCallbacksProcessed) return;
368
369 PromiseState_t s = state();
370 if (s == PromiseState_t::RESOLVED && mThenCallback) {
371 mThenCallback(mValue);
372 } else if (s == PromiseState_t::REJECTED && mCatchCallback) {
373 mCatchCallback(mError);
374 }
375
376 mCallbacksProcessed = true;
377 }
