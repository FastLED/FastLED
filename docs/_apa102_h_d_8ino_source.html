35#define NUM_LEDS 20
36// uint8_t DATA_PIN, uint8_t CLOCK_PIN,

42CRGB leds_hd[NUM_LEDS] = {0}; // HD mode implies gamma.
43CRGB leds[NUM_LEDS] = {0}; // Software gamma mode.
44
45// This is the regular gamma correction function that we used to have
46// to do. It's used here to showcase the difference between APA102HD
47// mode which does the gamma correction for you.
48CRGB software_gamma(const CRGB& in) {
49 CRGB out;
50 // dim8_raw are the old gamma correction functions.
51 out.r = dim8_raw(in.r);
52 out.g = dim8_raw(in.g);
53 out.b = dim8_raw(in.b);
54 return out;
55}

57void setup() {
58 delay(500); // power-up safety delay
59 // Two strips of LEDs, one in HD mode, one in software gamma mode.
60 FastLED.addLeds<APA102HD, STRIP_0_DATA_PIN, STRIP_0_CLOCK_PIN, RGB>(leds_hd, NUM_LEDS);
61 FastLED.addLeds<APA102, STRIP_1_DATA_PIN, STRIP_1_CLOCK_PIN, RGB>(leds, NUM_LEDS);
62}

64uint8_t wrap_8bit(int i) {
65 // Module % operator here wraps a large "i" so that it is
66 // always in [0, 255] range when returned. For example, if
67 // "i" is 256, then this will return 0. If "i" is 257
68 // then this will return 1. No matter how big the "i" is, the
69 // output range will always be [0, 255]
70 return i % 256;
71}

73void loop() {
74 // Draw a a linear ramp of brightnesses to showcase the difference between
75 // the HD and non-HD mode.
76 for (int i = 0; i < NUM_LEDS; i++) {
77 uint8_t brightness = map(i, 0, NUM_LEDS - 1, 0, 255);
78 CRGB c(brightness, brightness, brightness); // Just make a shade of white.
79 leds_hd[i] = c; // The APA102HD leds do their own gamma correction.
80 CRGB c_gamma_corrected = software_gamma(c);
81 leds[i] = c_gamma_corrected; // Set the software gamma corrected
82 // values to the other strip.
83 }
84 FastLED.show(); // All leds are now written out.
85 delay(8); // Wait 8 milliseconds until the next frame.
86}
