232 static CRGB computeAdjustment(uint8_t scale, const CRGB & colorCorrection, const CRGB & colorTemperature) {
233 #if defined(NO_CORRECTION) && (NO_CORRECTION==1)
234 return CRGB(scale,scale,scale);
235 #else
236 CRGB adj(0,0,0);
237
238 if(scale > 0) {
239 for(uint8_t i = 0; i < 3; ++i) {
240 uint8_t cc = colorCorrection.raw[i];
241 uint8_t ct = colorTemperature.raw[i];
242 if(cc > 0 && ct > 0) {
243 uint32_t work = (((uint32_t)cc)+1) * (((uint32_t)ct)+1) * scale;
244 work /= 0x10000L;
245 adj.raw[i] = work & 0xFF;
246 }
247 }
248 }
249
250 return adj;
251 #endif
252 }
