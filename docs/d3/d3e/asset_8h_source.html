50constexpr bool is_parent_at(const char* p, fl::size i) FL_NOEXCEPT {
51 // Segment start: either we're at the beginning, or preceded by a separator.
52 return (i == 0 || p[i - 1] == '/' || p[i - 1] == '\\')
