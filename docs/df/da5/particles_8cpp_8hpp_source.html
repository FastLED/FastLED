115void Particles1d::Particle::update(u32 now, u16 numLeds, float speedMultiplier, bool cyclical) {
116 if (!active) return;

118 if (power <= 0.0f) {
119 active = false;
120 return;

123 pos += baseVel * power * speedMultiplier;
