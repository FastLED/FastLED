453 FASTLED_FORCE_INLINE void setParity( u8 parity) FL_NOEXCEPT
454 {
455 u8 curparity = getParity();
456
457 if( parity == curparity) return;
458
459 if( parity ) {
460 // going 'up'
461 if( (b > 0) && (b < 255)) {
462 if( r == g && g == b) {
463 ++r;
464 ++g;
465 }
466 ++b;
467 } else if( (r > 0) && (r < 255)) {
468 ++r;
469 } else if( (g > 0) && (g < 255)) {
470 ++g;
471 } else {
472 if( r == g && g == b) {
473 r ^= 0x01;
474 g ^= 0x01;
475 }
476 b ^= 0x01;
477 }
478 } else {
479 // going 'down'
480 if( b > 1) {
481 if( r == g && g == b) {
482 --r;
483 --g;
484 }
485 --b;
486 } else if( g > 1) {
487 --g;
488 } else if( r > 1) {
489 --r;
490 } else {
491 if( r == g && g == b) {
492 r ^= 0x01;
493 g ^= 0x01;
494 }
495 b ^= 0x01;
496 }
497 }
