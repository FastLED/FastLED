179 // Calculate dt.
180 static uint32_t s_prev_time = 0;
181 uint32_t prev_time = 0;
182 FASTLED_UNUSED(prev_time); // actually used in perf tests.
183 uint32_t now_ms = millis();
184 uint32_t delta_ms = now_ms - s_prev_time;
185 s_prev_time = now_ms;
186
187 if (!is_debugging) {
188 if (Serial.available() > 0) {
189 int v = Serial.read();
190 if (v == 'd') {
191 is_debugging = true;
192 }
193 }
194 }
195
196 DbgDoSimulatedKeyboardPress();
197
198 const unsigned long start_time = millis();
199 // Each frame we call the midi processor 100 times to make sure that all notes
200 // are processed.
201 for (int i = 0; i < 100; ++i) {
202 MY_MIDI.read();
203 }
204
205 const unsigned long midi_time = millis() - start_time;
206
207 // Updates keyboard: releases sustained keys that.
208
209 const uint32_t keyboard_time_start = millis();
210
211 // This is kind of a hack... but give the keyboard a future time
212 // so that all keys just pressed get a value > 0 for their time
213 // durations.
214 keyboard.Update(now_ms + delta_ms, delta_ms);
215 const uint32_t keyboard_delta_time = millis() - keyboard_time_start;
216
217 ui_state ui_st = ui_update(now_ms, delta_ms);
218
219 //dprint("vis selector = ");
220 //dprintln(vis_state);
221
222
223 // These int values are for desting the performance of the
224 // app. If the app ever runs slow then set kShowFps to 1
225 // in the settings.h file.
226 const unsigned long start_painting = millis();
227 FASTLED_UNUSED(start_painting);
228
229
230 // Paints the keyboard using the led_rope.
231 Painter::VisState which_vis = Painter::VisState(ui_st.which_visualizer);
232 Painter::Paint(now_ms, delta_ms, which_vis, &keyboard, &led_rope);
233
234 const unsigned long paint_time = millis() - start_time;
235 const unsigned long total_time = midi_time + paint_time + keyboard_delta_time;
236
237 if (kShowFps) {
238 float fps = 1.0f/(float(total_time) / 1000.f);
239 Serial.print("fps - "); Serial.println(fps);
240 Serial.print("midi time - "); Serial.println(midi_time);
241 Serial.print("keyboard update time - "); Serial.println(keyboard_delta_time);
242 Serial.print("draw & paint time - "); Serial.println(paint_time);
243 }

245 EVERY_N_SECONDS(1) {
246 FASTLED_DBG("is_debugging = " << is_debugging);
247 }

250 FastLED.show();
251}
