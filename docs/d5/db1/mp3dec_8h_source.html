70/* determining MAINBUF_SIZE:
71 * max mainDataBegin = (2^9 - 1) bytes (since 9-bit offset) = 511
72 * max nSlots (concatenated with mainDataBegin bytes from before) = 1440 - 9 - 4 + 1 = 1428
73 * 511 + 1428 = 1939, round up to 1940 (4-byte align)
74 */
75#define MAINBUF_SIZE 1940
