74// Emulution layer to support RGBW leds on RGB controllers. This works by creating
75// a side buffer dedicated for the RGBW data. The RGB data is then converted to RGBW
76// and sent to the delegate controller for rendering as if it were RGB data.
77template <
78 typename CONTROLLER,
79 EOrder RGB_ORDER = GRB> // Default on WS2812>
80class RGBWEmulatedController
81 : public CPixelLEDController<RGB_ORDER, CONTROLLER::LANES_VALUE,
82 CONTROLLER::MASK_VALUE> {
83 public:
84 static const int LANES = CONTROLLER::LANES_VALUE;
85 static const uint32_t MASK = CONTROLLER::MASK_VALUE;
86
87 // The delegated controller must do no reordering.
88 static_assert(RGB == CONTROLLER::RGB_ORDER_VALUE);
89
90 RGBWEmulatedController(const Rgbw& rgbw = RgbwDefault()) {
91 this->setRgbw(rgbw);
92 };
93 ~RGBWEmulatedController() { delete[] mRGBWPixels; }
94
95 virtual void showPixels(PixelController<RGB_ORDER, LANES, MASK> &pixels) {
96 // Ensure buffer is large enough
97 ensureBuffer(pixels.size());
98 // This version sent down to the real controller.
99 PixelController<RGB, LANES, MASK> pixels_device(pixels);
100 pixels_device.mScale = CRGB(255, 255, 255); // No scaling because we do that.
101 pixels_device.mData = reinterpret_cast<uint8_t *>(mRGBWPixels);
102 pixels_device.mLen = mNumRGBWLeds;
103 pixels_device.mLenRemaining = mNumRGBWLeds;
104 uint8_t *data = reinterpret_cast<uint8_t *>(mRGBWPixels);
105 PixelIterator iterator = pixels.as_iterator(this->getRgbw());
106 while (iterator.has(1)) {
107 pixels.stepDithering();
108 iterator.loadAndScaleRGBW(data, data + 1, data + 2, data + 3);
109 data += 4;
110 iterator.advanceData();
111 }
112 // cast to base class to get around protected/private access issues
113 CPixelLEDController<RGB, LANES, MASK> &base = mController;
114 base.showPixels(pixels_device);
115 }
116
117 private:
118 // Needed by the interface.
119 void init() override {}
120
121 void ensureBuffer(int32_t num_leds) {
122 if (num_leds != mNumRGBLeds) {
123 mNumRGBLeds = num_leds;
124 // The delegate controller expects the raw pixel byte data in multiples of 3.
125 // In the case of src data not a multiple of 3, then we need to
126 // add pad bytes so that the delegate controller doesn't walk off the end
127 // of the array and invoke a buffer overflow panic.
128 mNumRGBWLeds = (num_leds * 4 + 2) / 3; // Round up to nearest multiple of 3
129 size_t extra = mNumRGBWLeds % 3 ? 1 : 0;
130 delete[] mRGBWPixels;
131 mRGBWPixels = new CRGB[mNumRGBWLeds + extra];
132 }
133 }
134
135 CRGB *mRGBWPixels = nullptr;
136 int32_t mNumRGBLeds = 0;
137 int32_t mNumRGBWLeds = 0;
138 CONTROLLER mController; // Real controller.
139};

288template <
289 uint8_t DATA_PIN, uint8_t CLOCK_PIN,
290 EOrder RGB_ORDER = RGB,
291 // APA102 has a bug where long strip can't handle full speed due to clock degredation.
292 // This only affects long strips, but then again if you have a short strip does 6 mhz actually slow
293 // you down? Probably not. And you can always bump it up for speed. Therefore we are prioritizing
294 // "just works" over "fastest possible" here.
295 // https://www.pjrc.com/why-apa102-leds-have-trouble-at-24-mhz/
296 uint32_t SPI_SPEED = DATA_RATE_MHZ(6),
297 FiveBitGammaCorrectionMode GAMMA_CORRECTION_MODE = kFiveBitGammaCorrectionMode_Null,
298 uint32_t START_FRAME = 0x00000000,
299 uint32_t END_FRAME = 0xFF000000
300>
301class APA102Controller : public CPixelLEDController<RGB_ORDER> {
302 typedef SPIOutput<DATA_PIN, CLOCK_PIN, SPI_SPEED> SPI;
303 SPI mSPI;
304
305 void startBoundary() {
306 mSPI.writeWord(START_FRAME >> 16);
307 mSPI.writeWord(START_FRAME & 0xFFFF);
308 }
309 void endBoundary(int nLeds) {
310 int nDWords = (nLeds/32);
311 const uint8_t b0 = uint8_t(END_FRAME >> 24 & 0x000000ff);
312 const uint8_t b1 = uint8_t(END_FRAME >> 16 & 0x000000ff);
313 const uint8_t b2 = uint8_t(END_FRAME >> 8 & 0x000000ff);
314 const uint8_t b3 = uint8_t(END_FRAME >> 0 & 0x000000ff);
315 do {
316 mSPI.writeByte(b0);
317 mSPI.writeByte(b1);
318 mSPI.writeByte(b2);
319 mSPI.writeByte(b3);
320 } while(nDWords--);
321 }
322
323 inline void writeLed(uint8_t brightness, uint8_t b0, uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
324#ifdef FASTLED_SPI_BYTE_ONLY
325 mSPI.writeByte(0xE0 | brightness);
326 mSPI.writeByte(b0);
327 mSPI.writeByte(b1);
328 mSPI.writeByte(b2);
329#else
330 uint16_t b = 0xE000 | (brightness << 8) | (uint16_t)b0;
331 mSPI.writeWord(b);
332 uint16_t w = b1 << 8;
333 w |= b2;
334 mSPI.writeWord(w);
335#endif
336 }
337
338 inline void write2Bytes(uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
339#ifdef FASTLED_SPI_BYTE_ONLY
340 mSPI.writeByte(b1);
341 mSPI.writeByte(b2);
342#else
343 mSPI.writeWord(uint16_t(b1) << 8 | b2);
344#endif
345 }
346
347public:
348 APA102Controller() {}
349
350 virtual void init() {
351 mSPI.init();
352 }
353
354protected:
356 virtual void showPixels(PixelController<RGB_ORDER> & pixels) {
357 PixelIterator iterator = pixels.as_iterator(this->getRgbw());
358 switch (GAMMA_CORRECTION_MODE) {
359 case kFiveBitGammaCorrectionMode_Null: {
360 showPixelsDefault(iterator);
361 break;
362 }
363 case kFiveBitGammaCorrectionMode_BitShift: {
364 showPixelsGammaBitShift(iterator);
365 break;
366 }
367 }
368 }

370private:
371
372 static inline void getGlobalBrightnessAndScalingFactors(
373 PixelIterator& pixels,
374 uint8_t* out_s0, uint8_t* out_s1, uint8_t* out_s2, uint8_t* out_brightness) {
375 uint8_t s0, s1, s2;
376 pixels.loadAndScaleRGB(&s0, &s1, &s2);
377#if FASTLED_USE_GLOBAL_BRIGHTNESS == 1
378 // This function is pure magic.
379 const uint16_t maxBrightness = 0x1F;
380 uint16_t brightness = ((((uint16_t)max(max(s0, s1), s2) + 1) * maxBrightness - 1) >> 8) + 1;
381 s0 = (maxBrightness * s0 + (brightness >> 1)) / brightness;
382 s1 = (maxBrightness * s1 + (brightness >> 1)) / brightness;
383 s2 = (maxBrightness * s2 + (brightness >> 1)) / brightness;
384#else
385 const uint8_t brightness = 0x1F;
386#endif
387 *out_s0 = s0;
388 *out_s1 = s1;
389 *out_s2 = s2;
390 *out_brightness = static_cast<uint8_t>(brightness);
391 }
392
393 // Legacy showPixels implementation.
394 inline void showPixelsDefault(PixelIterator& pixels) {
395 mSPI.select();
396 uint8_t s0, s1, s2, global_brightness;
397 getGlobalBrightnessAndScalingFactors(pixels, &s0, &s1, &s2, &global_brightness);
398 startBoundary();
399 while (pixels.has(1)) {
400 uint8_t c0, c1, c2;
401 pixels.loadAndScaleRGB(&c0, &c1, &c2);
402 writeLed(global_brightness, c0, c1, c2);
403 pixels.stepDithering();
404 pixels.advanceData();
405 }
406 endBoundary(pixels.size());
407
408 mSPI.waitFully();
409 mSPI.release();
410 }
411
412 inline void showPixelsGammaBitShift(PixelIterator& pixels) {
413 mSPI.select();
414 startBoundary();
415 while (pixels.has(1)) {
416 // Load raw uncorrected r,g,b values.
417 uint8_t brightness, c0, c1, c2; // c0-c2 is the RGB data re-ordered for pixel
418 pixels.loadAndScale_APA102_HD(&c0, &c1, &c2, &brightness);
419 writeLed(brightness, c0, c1, c2);
420 pixels.stepDithering();
421 pixels.advanceData();
422 }
423 endBoundary(pixels.size());
424 mSPI.waitFully();
425 mSPI.release();
426 }
