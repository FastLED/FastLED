547 inline void maximizeBrightness( uint8_t limit = 255 ) {
548 uint8_t max = red;
549 if( green > max) max = green;
550 if( blue > max) max = blue;
551
552 // stop div/0 when color is black
553 if(max > 0) {
554 uint16_t factor = ((uint16_t)(limit) * 256) / max;
555 red = (red * factor) / 256;
556 green = (green * factor) / 256;
557 blue = (blue * factor) / 256;
558 }

586 inline uint8_t getParity()
587 {
588 uint8_t sum = r + g + b;
589 return (sum & 0x01);

614 inline void setParity( uint8_t parity)
615 {
616 uint8_t curparity = getParity();
617
618 if( parity == curparity) return;
619
620 if( parity ) {
621 // going 'up'
622 if( (b > 0) && (b < 255)) {
623 if( r == g && g == b) {
624 ++r;
625 ++g;
626 }
627 ++b;
628 } else if( (r > 0) && (r < 255)) {
629 ++r;
630 } else if( (g > 0) && (g < 255)) {
631 ++g;
632 } else {
633 if( r == g && g == b) {
634 r ^= 0x01;
635 g ^= 0x01;
636 }
637 b ^= 0x01;
638 }
639 } else {
640 // going 'down'
641 if( b > 1) {
642 if( r == g && g == b) {
643 --r;
644 --g;
645 }
646 --b;
647 } else if( g > 1) {
648 --g;
649 } else if( r > 1) {
650 --r;
651 } else {
652 if( r == g && g == b) {
653 r ^= 0x01;
654 g ^= 0x01;
655 }
656 b ^= 0x01;
657 }
658 }
