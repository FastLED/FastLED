546 inline void maximizeBrightness( uint8_t limit = 255 ) {
547 uint8_t max = red;
548 if( green > max) max = green;
549 if( blue > max) max = blue;
550
551 // stop div/0 when color is black
552 if(max > 0) {
553 uint16_t factor = ((uint16_t)(limit) * 256) / max;
554 red = (red * factor) / 256;
555 green = (green * factor) / 256;
556 blue = (blue * factor) / 256;

586 {
587 uint8_t sum = r + g + b;
588 return (sum & 0x01);

613 inline void setParity( uint8_t parity)
614 {
615 uint8_t curparity = getParity();
616
617 if( parity == curparity) return;
618
619 if( parity ) {
620 // going 'up'
621 if( (b > 0) && (b < 255)) {
622 if( r == g && g == b) {
623 ++r;
624 ++g;
625 }
626 ++b;
627 } else if( (r > 0) && (r < 255)) {
628 ++r;
629 } else if( (g > 0) && (g < 255)) {
630 ++g;
631 } else {
632 if( r == g && g == b) {
633 r ^= 0x01;
634 g ^= 0x01;
635 }
636 b ^= 0x01;
637 }
638 } else {
639 // going 'down'
640 if( b > 1) {
641 if( r == g && g == b) {
642 --r;
643 --g;
644 }
645 --b;
646 } else if( g > 1) {
647 --g;
648 } else if( r > 1) {
649 --r;
650 } else {
651 if( r == g && g == b) {
652 r ^= 0x01;
653 g ^= 0x01;
654 }
655 b ^= 0x01;
656 }
657 }
