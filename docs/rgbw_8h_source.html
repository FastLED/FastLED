116FASTLED_FORCE_INLINE void rgb_2_rgbw(
120 RGBW_MODE mode, uint16_t w_color_temperature, uint8_t r, uint8_t g,
121 uint8_t b, uint8_t r_scale, uint8_t g_scale, uint8_t b_scale,
122 uint8_t *out_r, uint8_t *out_g, uint8_t *out_b, uint8_t *out_w) {
123 switch (mode) {
124 case kRGBWInvalid:
125 case kRGBWNullWhitePixel:
126 rgb_2_rgbw_null_white_pixel(w_color_temperature, r, g, b, r_scale,
127 g_scale, b_scale, out_r, out_g, out_b,
128 out_w);
129 return;
130 case kRGBWExactColors:
131 rgb_2_rgbw_exact(w_color_temperature, r, g, b, r_scale, g_scale,
132 b_scale, out_r, out_g, out_b, out_w);
133 return;
134 case kRGBWBoostedWhite:
135 rgb_2_rgbw_white_boosted(w_color_temperature, r, g, b, r_scale, g_scale,
136 b_scale, out_r, out_g, out_b, out_w);
137 return;
138 case kRGBWMaxBrightness:
139 rgb_2_rgbw_max_brightness(w_color_temperature, r, g, b, r_scale,
140 g_scale, b_scale, out_r, out_g, out_b, out_w);
141 return;
142 case kRGBWUserFunction:
143 rgb_2_rgbw_user_function(w_color_temperature, r, g, b, r_scale, g_scale,
144 b_scale, out_r, out_g, out_b, out_w);
145 return;
146 }
147 rgb_2_rgbw_null_white_pixel(w_color_temperature, r, g, b, r_scale, g_scale,
148 b_scale, out_r, out_g, out_b, out_w);
149}
