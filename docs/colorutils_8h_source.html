105template <typename T>
106void fill_gradient( T* targetArray,
107 uint16_t startpos, CHSV startcolor,
108 uint16_t endpos, CHSV endcolor,
109 TGradientDirectionCode directionCode = SHORTEST_HUES )
110{
111 // if the points are in the wrong order, straighten them
112 if( endpos < startpos ) {
113 uint16_t t = endpos;
114 CHSV tc = endcolor;
115 endcolor = startcolor;
116 endpos = startpos;
117 startpos = t;
118 startcolor = tc;
119 }
120
121 // If we're fading toward black (val=0) or white (sat=0),
122 // then set the endhue to the starthue.
123 // This lets us ramp smoothly to black or white, regardless
124 // of what 'hue' was set in the endcolor (since it doesn't matter)
125 if( endcolor.value == 0 || endcolor.saturation == 0) {
126 endcolor.hue = startcolor.hue;
127 }
128
129 // Similarly, if we're fading in from black (val=0) or white (sat=0)
130 // then set the starthue to the endhue.
131 // This lets us ramp smoothly up from black or white, regardless
132 // of what 'hue' was set in the startcolor (since it doesn't matter)
133 if( startcolor.value == 0 || startcolor.saturation == 0) {
134 startcolor.hue = endcolor.hue;
135 }
136
137 saccum87 huedistance87;
138 saccum87 satdistance87;
139 saccum87 valdistance87;
140
141 satdistance87 = (endcolor.sat - startcolor.sat) << 7;
142 valdistance87 = (endcolor.val - startcolor.val) << 7;
143
144 uint8_t huedelta8 = endcolor.hue - startcolor.hue;
145
146 if( directionCode == SHORTEST_HUES ) {
147 directionCode = FORWARD_HUES;
148 if( huedelta8 > 127) {
149 directionCode = BACKWARD_HUES;
150 }
151 }
152
153 if( directionCode == LONGEST_HUES ) {
154 directionCode = FORWARD_HUES;
155 if( huedelta8 < 128) {
156 directionCode = BACKWARD_HUES;
157 }
158 }
159
160 if( directionCode == FORWARD_HUES) {
161 huedistance87 = huedelta8 << 7;
162 }
163 else /* directionCode == BACKWARD_HUES */
164 {
165 huedistance87 = (uint8_t)(256 - huedelta8) << 7;
166 huedistance87 = -huedistance87;
167 }
168
169 uint16_t pixeldistance = endpos - startpos;
170 int16_t divisor = pixeldistance ? pixeldistance : 1;
171
172 #if FASTLED_USE_32_BIT_GRADIENT_FILL
173 // Use higher precision 32 bit math for new micros.
174 int32_t huedelta823 = (huedistance87 * 65536) / divisor;
175 int32_t satdelta823 = (satdistance87 * 65536) / divisor;
176 int32_t valdelta823 = (valdistance87 * 65536) / divisor;
177
178 huedelta823 *= 2;
179 satdelta823 *= 2;
180 valdelta823 *= 2;
181 uint32_t hue824 = static_cast<uint32_t>(startcolor.hue) << 24;
182 uint32_t sat824 = static_cast<uint32_t>(startcolor.sat) << 24;
183 uint32_t val824 = static_cast<uint32_t>(startcolor.val) << 24;
184 for( uint16_t i = startpos; i <= endpos; ++i) {
185 targetArray[i] = CHSV( hue824 >> 24, sat824 >> 24, val824 >> 24);
186 hue824 += huedelta823;
187 sat824 += satdelta823;
188 val824 += valdelta823;
189 }
190 #else
191 // Use 8-bit math for older micros.
192 saccum87 huedelta87 = huedistance87 / divisor;
193 saccum87 satdelta87 = satdistance87 / divisor;
194 saccum87 valdelta87 = valdistance87 / divisor;
195
196 huedelta87 *= 2;
197 satdelta87 *= 2;
198 valdelta87 *= 2;
199
200 accum88 hue88 = startcolor.hue << 8;
201 accum88 sat88 = startcolor.sat << 8;
202 accum88 val88 = startcolor.val << 8;
203 for( uint16_t i = startpos; i <= endpos; ++i) {
204 targetArray[i] = CHSV( hue88 >> 8, sat88 >> 8, val88 >> 8);
205 hue88 += huedelta87;
206 sat88 += satdelta87;
207 val88 += valdelta87;
208 }
209 #endif // defined(__AVR__)
210}
