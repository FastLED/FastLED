50// Cycle r,g,b,w. Red will blink once, green twice, ... white 4 times.
51void loop() {
52 uint32_t ms = millis();
53 static size_t frame_count = 0;
54 int frame_cycle = frame_count % 4;
55 frame_count++;
56
57 CRGB pixel;
58 switch (frame_cycle) {
59 case 0:
60 pixel = CRGB::Red;
61 break;
62 case 1:
63 pixel = CRGB::Green;
64 break;
65 case 2:
66 pixel = CRGB::Blue;
67 break;
68 case 3:
69 pixel = CRGB::White;
70 break;
71 }
72
73 for (int i = -1; i < frame_cycle; ++i) {
74 fillAndShow(pixel);
75 delay(200);
76 fillAndShow(CRGB::Black);
77 delay(200);
78 }
79 delay(1000);
80}
