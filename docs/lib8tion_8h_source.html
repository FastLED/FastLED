586LIB8STATIC uint8_t ease8InOutQuad( uint8_t i)
587{

590 j = 255 - j;

601// This AVR asm version of ease8InOutQuad preserves one more
602// low-bit of precision than the C version, and is also slightly
603// smaller and faster.
604LIB8STATIC uint8_t ease8InOutQuad(uint8_t val) {
605 uint8_t j=val;
606 asm volatile (
607 "sbrc %[val], 7 \n"
608 "com %[j] \n"
609 "mul %[j], %[j] \n"
610 "add r0, %[j] \n"
611 "ldi %[j], 0 \n"
612 "adc %[j], r1 \n"
613 "lsl r0 \n" // carry = high bit of low byte of mul product
614 "rol %[j] \n" // j = (j * 2) + carry // preserve add'l bit of precision
615 "sbrc %[val], 7 \n"
616 "com %[j] \n"
617 "clr __zero_reg__ \n"
618 : [j] "+&a" (j)
619 : [val] "a" (val)
620 : "r0", "r1"
621 );

648LIB8STATIC fract8 ease8InOutCubic( fract8 i)
649{
650 uint8_t ii = scale8_LEAVING_R1_DIRTY( i, i);
651 uint8_t iii = scale8_LEAVING_R1_DIRTY( ii, i);
652
653 uint16_t r1 = (3 * (uint16_t)(ii)) - ( 2 * (uint16_t)(iii));
654
655 /* the code generated for the above *'s automatically
656 cleans up R1, so there's no need to explicitily call
657 cleanup_R1(); */
658
659 uint8_t result = r1;
660
661 // if we got "256", return 255:
662 if( r1 & 0x100 ) {
663 result = 255;
664 }
665 return result;
666}

677LIB8STATIC fract8 ease8InOutApprox( fract8 i)
678{
679 if( i < 64) {
680 // start with slope 0.5
681 i /= 2;
682 } else if( i > (255 - 64)) {
683 // end with slope 0.5
684 i = 255 - i;
685 i /= 2;
686 i = 255 - i;

689 i -= 64;
690 i += (i / 2);
691 i += 32;
692 }
693
694 return i;

954LIB8STATIC uint16_t beat88( accum88 beats_per_minute_88, uint32_t timebase = 0)
955{
956 // BPM is 'beats per minute', or 'beats per 60000ms'.
957 // To avoid using the (slower) division operator, we
958 // want to convert 'beats per 60000ms' to 'beats per 65536ms',
959 // and then use a simple, fast bit-shift to divide by 65536.
960 //
961 // The ratio 65536:60000 is 279.620266667:256; we'll call it 280:256.
962 // The conversion is accurate to about 0.05%, more or less,
963 // e.g. if you ask for "120 BPM", you'll get about "119.93".
964 return (((GET_MILLIS()) - timebase) * beats_per_minute_88 * 280) >> 16;
965}

970LIB8STATIC uint16_t beat16( accum88 beats_per_minute, uint32_t timebase = 0)
